# Function to create model, required for KerasClassifier
def create_model(optimizer='adam'):
    # create model
    model = Sequential()
    model.add(Dense(100, input_shape = (1024, )))
    model.add(BatchNormalization())                    
    model.add(Activation('relu'))    
    model.add(Dense(100))
    model.add(BatchNormalization())                    
    model.add(Activation('relu'))    
    model.add(Dense(10))
    model.add(Activation('softmax'))
    # Compile model
    model.compile(loss='binary_crossentropy', optimizer=optimizer, metrics=['accuracy'])
    return model

from keras.constraints import maxnorm
# Function to create model, required for KerasClassifier
def create_model(dropout_rate=0.0, weight_constraint=0):
    # create model
    model = Sequential()
    model.add(Dense(100, input_shape = (1024, ), kernel_initializer='zero', kernel_constraint=maxnorm(weight_constraint)))
    model.add(BatchNormalization())                    
    model.add(Dropout(dropout_rate))
    model.add(Activation('relu'))    
    model.add(Dense(100, kernel_initializer='zero', kernel_constraint=maxnorm(weight_constraint)))
    model.add(BatchNormalization())                    
    model.add(Dropout(dropout_rate))
    model.add(Activation('relu'))    
    model.add(Dense(100, kernel_initializer='zero', kernel_constraint=maxnorm(weight_constraint)))
    model.add(BatchNormalization())                    
    model.add(Dropout(dropout_rate))
    model.add(Activation('relu'))    
    model.add(Dense(10, kernel_initializer='zero', kernel_constraint=maxnorm(weight_constraint)))
    model.add(Activation('softmax'))
    # Compile model
    model.compile(loss='binary_crossentropy', optimizer='adam', metrics=['accuracy'])
    return model
